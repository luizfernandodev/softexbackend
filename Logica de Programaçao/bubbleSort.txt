   // Seção de Declarações das variáveis
   v: vetor [1..10] de Inteiro
   i, j, aux: Inteiro

Inicio
   // Seção de Comandos, procedimento, funções, operadores, etc...
   para i <- 1 ate 10 faca
      Escreva("Digite um valor: ")
      leia(v[i])
   fimpara

   para i <- 1 ate 9 faca
      para j <- i + 1 ate 10 faca
         se (v[i] > v[j]) entao
            aux <- v[i]
            v[i] <- v[j]
            v[j] <- aux
         fimse
      fimpara
   fimpara
